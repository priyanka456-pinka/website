#include <stdio.h>

// Function to convert decimal to binary using recursion
void decimalToBinary(int n) {
if (n > 1) {
decimalToBinary(n / 2); // Recursive call with n divided by 2
}
printf("%d", n % 2); // Print the remainder (0 or 1)
}

int main() {
int num;
printf("Enter a decimal number to convert to binary: ");
scanf("%d", &num);

if (num < 0) {
printf("Please enter a non-negative integer.\n");
} else {
printf("Binary representation of %d is: ", num);
decimalToBinary(num);
printf("\n");
}

return 0;
}

OUTPUT

Enter a decimal number to convert to binary: 18
Binary representation of 18 is: 10010
