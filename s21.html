#include <stdio.h>
#include <string.h>
// Function to swap characters at position i and j
void swap(char *x, char *y) {
char temp;
temp = *x;
*x = *y;
*y = temp;
}
// Recursive function to generate permutations
void permute(char *str, int left, int right) {
if (left == right) {
printf("%s\n", str); // Print the current permutation
} else {
for (int i = left; i <= right; i++) {
swap((str + left), (str + i)); // Swap the fixed character
permute(str, left + 1, right); // Recur for the rest of the string
swap((str + left), (str + i)); // Backtrack to the original string
}
}
}
int main() {
char str[] = "ABCDE"; // String to permute
int n = strlen(str); // Length of the string
printf("All permutations of the string \"%s\":\n", str);
permute(str, 0, n - 1); // Call the permutation function
return 0;
}
