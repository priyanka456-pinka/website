#include <stdio.h>
#include <string.h>

int linearSearch(char str[], char target) {
for (int i = 0; i < strlen(str); i++) {
if (str[i] == target) {
return i;
}
}
return -1;
}

int main() {
char str[100];
char target;

printf("Enter a string: ");
fgets(str, sizeof(str), stdin);

str[strcspn(str, "\n")] = 0;

printf("Enter a character to search for: ");
scanf(" %c", &target);

int index = linearSearch(str, target);

if (index != -1) {
printf("Character '%c' found at index: %d\n", target, index);
} else {
printf("Character '%c' not found in the string.\n", target);
}

return 0;
}
