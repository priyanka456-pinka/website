#include <stdio.h>

int main() {
int n;

printf("Enter the number of elements in the array: ");
scanf("%d", &n);

int arr[n];

printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
printf("Element %d: ", i + 1);
scanf("%d", &arr[i]);
}

printf("Duplicate values in the array are:\n");
int foundDuplicate = 0;
for (int i = 0; i < n; i++) {
for (int j = i + 1; j < n; j++) {
if (arr[i] == arr[j]) {
printf("%d ", arr[i]);
foundDuplicate = 1;
break;
}
}
}

if (!foundDuplicate) {
printf("No duplicate values found.");
}
printf("\n");

return 0;
}
