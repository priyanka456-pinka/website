#include <stdio.h>
#include <string.h>
#include <time.h>
#define MAX_STRINGS 100
void merge(char arr[][100], int left, int mid, int right) {
int i, j, k;
int n1 = mid - left + 1;
int n2 = right - mid;
char L[MAX_STRINGS][100], R[MAX_STRINGS][100];
for (i = 0; i < n1; i++)
strcpy(L[i], arr[left + i]);
for (j = 0; j < n2; j++)
strcpy(R[j], arr[mid + 1 + j]);
i = 0;
j = 0;
k = left;
while (i < n1 && j < n2) {
if (strcmp(L[i], R[j]) <= 0) {
strcpy(arr[k], L[i]);
i++;
} else {
strcpy(arr[k], R[j]);
j++;
}
k++;
}
while (i < n1) {
strcpy(arr[k], L[i]);
i++;
k++;
}
while (j < n2) {
strcpy(arr[k], R[j]);
j++;
k++;
}
}
void mergeSort(char arr[][100], int n) {
int curr_size;
int left_start;
for (curr_size = 1; curr_size <= n - 1; curr_size = 2 * curr_size) {
for (left_start = 0; left_start < n - 1; left_start += 2 * curr_size) {
int mid = left_start + curr_size - 1;
int right_end = left_start + 2 * curr_size - 1;
if (right_end >= n)
right_end = n - 1;
if (mid < right_end)
merge(arr, left_start, mid, right_end);
}
}
}
int main() {
char arr[MAX_STRINGS][100];
int n;
clock_t start, end;
printf("Enter the number of strings: ");
scanf("%d", &n);
getchar();
printf("Enter %d strings:\n", n);
for (int i = 0; i < n; i++) {
fgets(arr[i], sizeof(arr[i]), stdin);
arr[i][strcspn(arr[i], "\n")] = '\0';
}
start = clock();
mergeSort(arr, n);
end = clock();
printf("\nSorted strings:\n");
for (int i = 0; i < n; i++) {
printf("%s\n", arr[i]);
}
double time_taken = (double)(end - start) / CLOCKS_PER_SEC;
printf("Time taken for sorting: %f seconds\n", time_taken);
printf("Best case time complexity: O(n log n)\n");
printf("Worst case time complexity: O(n log n)\n");
return 0;
}
