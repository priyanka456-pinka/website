#include <stdio.h>
void findLocalMinima(int arr[], int size) {
if (size < 3) {
printf("Array should have at least 3 elements to find local minima.\n");
return;
}
printf("Local minima are at the following indices and values:\n");
for (int i = 1; i < size - 1; i++) {
if (arr[i] < arr[i - 1] && arr[i] < arr[i + 1]) {
printf("Index: %d, Value: %d\n", i, arr[i]);
}
}
}
int main() {
int n;
printf("Enter the number of elements in the array: ");
scanf("%d", &n);
int arr[n];
printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &arr[i]);
}
findLocalMinima(arr, n);
return 0;
}
