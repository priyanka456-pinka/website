#include <stdio.h>

int main() {
char str1[100], str2[100];
int i = 0;

// Taking input for the first string
printf("Enter a string: ");
fgets(str1, sizeof(str1), stdin);

// Copying the string from str1 to str2
while (str1[i] != '\0') {
str2[i] = str1[i];
i++;
}

// Add the null terminator at the end of str2
str2[i] = '\0';

// Output the copied string
printf("Copied string: %s", str2);

return 0;
}
