#include <stdio.h>
int isPalindrome(int num) {
int reversed = 0, original = num;
while (num > 0) {
int digit = num % 10;
reversed = reversed * 10 + digit;
num /= 10;
}
return original == reversed;
}
int main() {
int n;
printf("Enter the number of elements in the array: ");
scanf("%d", &n);
int arr[n];
printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &arr[i]);
}
printf("Palindrome numbers in the array are:\n");
int foundPalindrome = 0;
for (int i = 0; i < n; i++) {
if (isPalindrome(arr[i])) {
printf("%d ", arr[i]);
foundPalindrome = 1;
}
}
if (!foundPalindrome) {
printf("No palindrome numbers found in the array.\n");
} else {
printf("\n");
}
return 0;
}
