#include <stdio.h>
int main() {
char str[100]; // Array to hold the input string
float number = 0; // Variable to hold the converted float value
int i = 0; // Loop counter
int sign = 1; // Variable to store the sign of the number (1 for positive, -1 for negative)
int decimalPoint = 0; // To indicate if we have encountered a decimal point
float decimalPlace = 1; // To track the place value for the fractional part
// Input: string from the user
printf("Enter a string to convert to a float: ");
fgets(str, sizeof(str), stdin); // Use fgets to read a line of text
// Check for a negative sign
if (str[0] == '-') {
sign = -1; // Set sign to -1 for negative numbers
i++; // Start from the next character
} else if (str[0] == '+') {
i++; // Ignore the plus sign and start from the next character
}
// Convert each character to a float
for (; str[i] != '\0'; i++) {
if (str[i] >= '0' && str[i] <= '9') {
// If it's a digit, calculate the float value
if (decimalPoint == 0) {
// Integer part
number = number * 10 + (str[i] - '0');
} else {
// Fractional part
decimalPlace *= 10;
number += (str[i] - '0') / decimalPlace;
}
} else if (str[i] == '.') {
// Set decimalPoint to indicate we've encountered the decimal
decimalPoint = 1;
} else if (str[i] == '\n') {
// Stop at newline character (if using fgets)
break;
} else {
// Handle non-numeric characters
printf("Error: Invalid character '%c' in input.\n", str[i]);
return 1; // Exit with error code
}
}
// Apply the sign to the number
number *= sign;
// Output the result
printf("The float value is: %f\n", number);
return 0;
}
