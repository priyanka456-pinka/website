CODE

#include <stdio.h>
#include <stdlib.h>

int main() {
char str[100];
int i;

// Get the string input from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin);

// Calculate and print the difference in ASCII values between consecutive characters
for (i = 0; str[i] != '\0' && str[i+1] != '\0'; i++) {
int current_char = str[i];
int next_char = str[i + 1];

// Calculate the absolute difference
int difference = abs(next_char - current_char);

// Print the difference
printf("Difference between '%c' and '%c' is %d\n", str[i], str[i + 1], difference);
}

return 0;
}
