struct TreeNode {
int val;
struct TreeNode *left, *right;
};
struct TreeNode* createNode(int value) {
struct TreeNode* newNode = (struct TreeNode*)malloc(sizeof(struct TreeNode));
newNode->val = value;
newNode->left = newNode->right = NULL;
return newNode;
}
struct TreeNode* createStrictlyBinaryTree() {
int value;
char choice;
printf("Enter value for the node: ");
scanf("%d", &value);
struct TreeNode* root = createNode(value);
printf("Does the node with value %d have left and right children? (y/n): ", value);
scanf(" %c", &choice);
if (choice == 'y' || choice == 'Y') {
printf("Creating left child of node with value %d.\n", value);
root->left = createStrictlyBinaryTree();
printf("Creating right child of node with value %d.\n", value);
root->right = createStrictlyBinaryTree();
}
return root;
}
void inorderTraversal(struct TreeNode* root) {
if (!root) return;
inorderTraversal(root->left);
printf("%d ", root->val);
inorderTraversal(root->right);
}
int main() {
printf("Create a strictly binary tree.\n");
struct TreeNode* root = createStrictlyBinaryTree();
printf("Inorder traversal of the created strictly binary tree: ");
inorderTraversal(root);
printf("\n");
return 0;
}
