#include <stdio.h>
#include <string.h>

// Function to compare two strings
int compareStrings(char *str1, char *str2) {
int i = 0;
while (str1[i] != '\0' && str2[i] != '\0') {
if (str1[i] != str2[i]) {
return 0; // Strings are not equal
}
i++;
}
// If both strings reached the end, they are equal
return (str1[i] == '\0' && str2[i] == '\0');
}

// Function to replace a word in a sentence
void replaceWord(char *sentence, char *target, char *replacement, char *result) {
int i = 0, j = 0, k = 0;
char temp[100]; // Temporary array to store each word
while (sentence[i] != '\0') {
// Extract each word from the sentence
j = 0;
while (sentence[i] != ' ' && sentence[i] != '\n' && sentence[i] != '\0') {
temp[j++] = sentence[i++];
}
temp[j] = '\0'; // Null-terminate the word
// Compare the extracted word with the target word
if (compareStrings(temp, target)) {
// If the word matches the target, add the replacement word to the result
j = 0;
while (replacement[j] != '\0') {
result[k++] = replacement[j++];
}
} else {
// Otherwise, add the original word to the result
j = 0;
while (temp[j] != '\0') {
result[k++] = temp[j++];
}
}
// Add a space after the word (if there is a space in the original sentence)
if (sentence[i] == ' ') {
result[k++] = ' ';
i++;
}
}
result[k] = '\0'; // Null-terminate the result string
}

int main() {
char sentence[1000]; // Array to hold the input sentence
char target[100]; // Array to hold the word to be replaced
char replacement[100]; // Array to hold the replacement word
char result[1000]; // Array to hold the final result
// Input: Get the sentence from the user
printf("Enter a sentence: ");
fgets(sentence, sizeof(sentence), stdin); // Read the input sentence
// Input: Get the word to be replaced
printf("Enter the word to be replaced: ");
fgets(target, sizeof(target), stdin); // Read the target word
// Remove the newline character from the target if it exists
int i = 0;
while (target[i] != '\0') {
if (target[i] == '\n') {
target[i] = '\0';
break;
}
i++;
}
// Input: Get the replacement word
printf("Enter the replacement word: ");
fgets(replacement, sizeof(replacement), stdin); // Read the replacement word
// Remove the newline character from the replacement if it exists
i = 0;
while (replacement[i] != '\0') {
if (replacement[i] == '\n') {
replacement[i] = '\0';
break;
}
i++;
}
// Replace the target word with the replacement word in the sentence
replaceWord(sentence, target, replacement, result);
// Output the final result
printf("The modified sentence is: %s\n", result);
return 0;
}
