#include <stdio.h>
#include <string.h>
#include <time.h>
#define MAX_STRINGS 100
#define MAX_LENGTH 100
void recursiveInsert(char arr[][MAX_LENGTH], int n) {
if (n <= 1) {
return;
}
recursiveInsert(arr, n - 1);
char last[MAX_LENGTH];
strcpy(last, arr[n - 1]);
int j = n - 2;
while (j >= 0 && strcmp(arr[j], last) > 0) {
strcpy(arr[j + 1], arr[j]);
j--;
}
strcpy(arr[j + 1], last);
}
void displayArray(char arr[][MAX_LENGTH], int n) {
for (int i = 0; i < n; i++) {
printf("%s\n", arr[i]);
}
}
int main() {
char arr[MAX_STRINGS][MAX_LENGTH];
int n;
printf("Enter the number of strings: ");
scanf("%d", &n);
getchar();
printf("Enter %d names:\n", n);
for (int i = 0; i < n; i++) {
fgets(arr[i], MAX_LENGTH, stdin);
arr[i][strcspn(arr[i], "\n")] = 0;
}
clock_t start = clock();
recursiveInsert(arr, n);
clock_t end = clock();
double time_taken = (double)(end - start) / CLOCKS_PER_SEC;
printf("Sorted names:\n");
displayArray(arr, n);
printf("Time taken for sorting: %f seconds\n", time_taken);
printf("Best case time complexity: O(n)\n");
printf("Worst case time complexity: O(n^2)\n");
return 0;
}
