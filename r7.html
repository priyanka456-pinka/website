#include <stdio.h>

// Function to calculate the sum of digits using recursion
int sumOfDigits(int n) {
// Base case: if n is 0, return 0
if (n == 0) {
return 0;
} else {
// Recursive case: add the last digit to the sum of the remaining digits
return (n % 10) + sumOfDigits(n / 10);
}
}

int main() {
int num;

printf("Enter a number: ");
scanf("%d", &num);

// Check if the number is negative
if (num < 0) {
num = -num; // Convert to positive for summing digits
}

int result = sumOfDigits(num);
printf("The sum of the digits of %d is: %d\n", num, result);

return 0;
}

OUTPUT

Enter a number: 12345
The sum of the digits of 12345 is: 15
