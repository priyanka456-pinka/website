#include <stdio.h>

int main() {
char str[100];
int i;

// Taking input from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin);

// Convert the first character to uppercase if it's a lowercase letter
if (str[0] >= 'a' && str[0] <= 'z') {
str[0] = str[0] - 32; // Convert to uppercase
}

// Convert the rest of the characters to lowercase
for (i = 1; str[i] != '\0'; i++) {
if (str[i] >= 'A' && str[i] <= 'Z') {
str[i] = str[i] + 32; // Convert to lowercase
}
}

// Output the converted string
printf("Sentence case string: %s", str);

return 0;
}

Sentence case string: My name is priyanka
