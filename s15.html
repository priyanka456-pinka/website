#include <stdio.h>

int main() {
char str[100]; // Array to hold the string
char ch; // Character to search for
int found = 0; // Flag to check if character is found
int i; // Loop counter

// Input: string from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin); // Use fgets to read a line of text

// Input: character to search for
printf("Enter a character to search: ");
scanf(" %c", &ch); // Note the space before %c to skip any newline character

// Search for the character in the string
printf("Character '%c' found at positions: ", ch);
for (i = 0; str[i] != '\0'; i++) {
if (str[i] == ch) {
printf("%d ", i); // Print the index where the character is found
found = 1; // Set flag to 1 if found
}
}

// If the character was not found, inform the user
if (!found) {
printf("not found in the string.");
}

printf("\n");
return 0;
}
