#include <stdio.h>
// Function to compare two strings
int compareStrings(char *str1, char *str2) {
int i = 0;
while (str1[i] != '\0' && str2[i] != '\0') {
if (str1[i] != str2[i]) {
return 0; // Strings are not equal
}
i++;
}
// If both strings reached the end, they are equal
return (str1[i] == '\0' && str2[i] == '\0');
}
// Function to count occurrences of a word in a sentence
int countOccurrences(char *sentence, char *word) {
int count = 0;
int i = 0, j = 0;
char temp[100]; // Temporary array to store each word
while (sentence[i] != '\0') {
// Extract each word from the sentence
j = 0;
while (sentence[i] != ' ' && sentence[i] != '\n' && sentence[i] != '\0') {
temp[j++] = sentence[i++];
}
temp[j] = '\0'; // Null-terminate the word
// Compare the extracted word with the target word
if (compareStrings(temp, word)) {
count++;
}
// Skip any spaces
while (sentence[i] == ' ') {
i++;
}
}
return count;
}
int main() {
char sentence[1000]; // Array to hold the input sentence
char word[100]; // Array to hold the target word
int occurrences;
// Input: Get the sentence from the user
printf("Enter a sentence: ");
fgets(sentence, sizeof(sentence), stdin); // Read the input sentence
// Input: Get the word to search for
printf("Enter the word to count: ");
fgets(word, sizeof(word), stdin); // Read the input word
// Remove the newline character from the word if it exists
int i = 0;
while (word[i] != '\0') {
if (word[i] == '\n') {
word[i] = '\0';
break;
}
i++;
}
// Count occurrences of the word in the sentence
occurrences = countOccurrences(sentence, word);
// Output the result
printf("The word \"%s\" occurs %d times in the sentence.\n", word, occurrences);
return 0;
}
