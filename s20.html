#include <stdio.h>
int main() {
char str[1000]; // Array to hold the input string
int charCount = 0; // Count of characters
int wordCount = 0; // Count of words
int spaceCount = 0; // Count of spaces
int lineCount = 0; // Count of lines
int i = 0; // Loop counter
// Input: string from the user
printf("Enter a string (max 1000 characters, press Enter to finish): \n");
fgets(str, sizeof(str), stdin); // Use fgets to read a line of text
// Count characters, words, spaces, and lines
for (i = 0; str[i] != '\0'; i++) {
// Count characters
if (str[i] != '\n') {
charCount++;
}
// Count spaces
if (str[i] == ' ') {
spaceCount++;
}
// Count lines
if (str[i] == '\n') {
lineCount++;
}
// Count words (a word is defined as a sequence of characters separated by spaces)
if ((i == 0 && str[i] != ' ') || (str[i] != ' ' && str[i - 1] == ' ')) {
wordCount++;
}
}
// If the string does not end with a newline, count the last line
if (charCount > 0 && str[i - 1] != '\n') {
lineCount++;
}
// Output the results
printf("Number of characters: %d\n", charCount);
printf("Number of words: %d\n", wordCount);
printf("Number of spaces: %d\n", spaceCount);
printf("Number of lines: %d\n", lineCount);
return 0;
}
