#include <stdio.h>

// Function to find the smallest element in the array using recursion
int findSmallest(int arr[], int n) {
// Base case: if there's only one element, return it
if (n == 1) {
return arr[0];
}

// Recursive case: compare the last element with the smallest of the rest
int smallestOfRest = findSmallest(arr, n - 1);
return (arr[n - 1] < smallestOfRest) ? arr[n - 1] : smallestOfRest;
}

int main() {
int n;

printf("Enter the number of elements in the array: ");
scanf("%d", &n);

if (n <= 0) {
printf("Please enter a positive integer.\n");
return 1; // Exit if the input is invalid
}

int arr[n];
printf("Enter %d integers:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &arr[i]);
}

int smallest = findSmallest(arr, n);
printf("The smallest element in the array is: %d\n", smallest);

return 0;
}
