CODE

#include <stdio.h>

int main() {
char str[100]; // Array to hold the string
char ch; // Character to search for
int index = -1; // Variable to store the index of the character
int i; // Loop counter

// Input: string from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin); // Use fgets to read a line of text

// Input: character to search for
printf("Enter a character to search: ");
scanf(" %c", &ch); // Note the space before %c to skip any newline character

// Search for the character in the string
for (i = 0; str[i] != '\0'; i++) {
if (str[i] == ch) {
index = i; // Store the index where the character is found
break; // Exit the loop after finding the first occurrence
}
}

// Output the result
printf("Character '%c' found at index: %d\n", ch, index);

return 0;
}
