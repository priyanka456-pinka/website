#include <stdio.h>
#define MAX_SIZE 100 // Maximum size of the array
int main() {
int arr[MAX_SIZE]; // Array to hold the user input
int n, i, j, count, duplicates[MAX_SIZE], dupCount = 0;
printf("Enter the number of elements in the array (max %d): ", MAX_SIZE);
scanf("%d", &n);
if (n > MAX_SIZE) {
printf("Number of elements exceeds maximum size of %d.\n", MAX_SIZE);
return 1;
}
printf("Enter %d elements:\n", n);
for (i = 0; i < n; i++) {
scanf("%d", &arr[i]);
}
for (i = 0; i < n; i++) {
count = 1;
for (j = i + 1; j < n; j++) {
if (arr[i] == arr[j]) {
count++;
arr[j] = -1;
}
}
if (count > 1 && arr[i] != -1) {
duplicates[dupCount++] = arr[i];
}
}
if (dupCount > 0) {
printf("Duplicate values in the array are:\n");
for (i = 0; i < dupCount; i++) {
printf("%d\n", duplicates[i]);
}
} else {
printf("No duplicate values found in the array.\n");
}
return 0;
}
