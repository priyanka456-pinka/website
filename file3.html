#include <stdio.h>
int main() {
int matrix1[2][2], matrix2[2][2], difference[2][2];
int i, j;

// Input the first matrix
printf("Enter elements of the first 2x2 matrix:\n");
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
printf("Element [%d][%d]: ", i, j);
scanf("%d", &matrix1[i][j]);
}
}

// Input the second matrix
printf("Enter elements of the second 2x2 matrix:\n");
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
printf("Element [%d][%d]: ", i, j);
scanf("%d", &matrix2[i][j]);
}
}

// Subtract the matrices
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
difference[i][j] = matrix1[i][j] - matrix2[i][j];
}
}

// Display the result
printf("\nThe difference of the two matrices is:\n");
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
printf("%d ", difference[i][j]);
}
printf("\n");
}

return 0;
}
