#include <stdio.h>
#include <string.h>

#define MAX_ENTRIES 10
#define MAX_WORD_LENGTH 50
#define MAX_MEANING_LENGTH 100

// Structure to represent a dictionary entry
typedef struct {
char word[MAX_WORD_LENGTH];
char meaning[MAX_MEANING_LENGTH];
} DictionaryEntry;

// Global array to hold dictionary entries
DictionaryEntry dictionary[MAX_ENTRIES] = {
{"Apple", "A fruit that is usually red, green, or yellow."},
{"Book", "A set of written or printed pages, bound together."},
{"Car", "A road vehicle, typically with four wheels, powered by an engine."},
{"Dog", "A domesticated carnivorous mammal that typically has a long snout."},
{"Elephant", "A large mammal with a trunk, native to Africa and Asia."},
{"Forest", "A large area covered chiefly with trees and undergrowth."},
{"Guitar", "A musical instrument with six strings played by strumming or plucking."},
{"House", "A building for human habitation."},
{"Internet", "A global network connecting millions of computers for communication."},
{"Jacket", "An outer garment extending to the waist or hips."}
};

int currentEntries = MAX_ENTRIES; // Current number of entries

// Function to create a new dictionary entry
void createEntry() {
if (currentEntries >= MAX_ENTRIES) {
printf("Dictionary is full! Cannot add more entries.\n");
return;
}
printf("Enter the word: ");
scanf("%s", dictionary[currentEntries].word);
getchar(); // Clear the newline character from input buffer
printf("Enter the meaning: ");
fgets(dictionary[currentEntries].meaning, MAX_MEANING_LENGTH, stdin);
// Remove newline character from meaning
dictionary[currentEntries].meaning[strcspn(dictionary[currentEntries].meaning, "\n")] = '\0';
currentEntries++;
printf("Entry added successfully!\n");
}

// Function to delete a dictionary entry
void deleteEntry() {
char word[MAX_WORD_LENGTH];
printf("Enter the word to delete: ");
scanf("%s", word);
int found = -1;
for (int i = 0; i < currentEntries; i++) {
if (strcmp(dictionary[i].word, word) == 0) {
found = i;
break;
}
}
if (found == -1) {
printf("Word not found in the dictionary!\n");
return;
}
// Shift entries to remove the deleted word
for (int i = found; i < currentEntries - 1; i++) {
dictionary[i] = dictionary[i + 1];
}
currentEntries--;
printf("Entry deleted successfully!\n");
}

// Function to search for a word in the dictionary
void searchEntry() {
char word[MAX_WORD_LENGTH];
printf("Enter the word to search: ");
scanf("%s", word);
for (int i = 0; i < currentEntries; i++) {
if (strcmp(dictionary[i].word, word) == 0) {
printf("Meaning of '%s': %s\n", word, dictionary[i].meaning);
return;
}
}
printf("Word not found in the dictionary!\n");
}

// Function to display all dictionary entries
void displayDictionary() {
if (currentEntries == 0) {
printf("The dictionary is empty!\n");
return;
}
printf("Dictionary entries:\n");
for (int i = 0; i < currentEntries; i++) {
printf("%d. %s: %s\n", i + 1, dictionary[i].word, dictionary[i].meaning);
}
}

int main() {
int choice;
while (1) {
// Menu options
printf("\nDictionary Operations:\n");
printf("1. Create a new entry\n");
printf("2. Delete an entry\n");
printf("3. Search for a word\n");
printf("4. Display all entries\n");
printf("5. Exit\n");
printf("Enter your choice: ");
scanf("%d", &choice);
switch (choice) {
case 1:
createEntry();
break;
case 2:
deleteEntry();
break;
case 3:
searchEntry();
break;
case 4:
displayDictionary();
break;
case 5:
printf("Exiting the program. Goodbye!\n");
return 0;
default:
printf("Invalid choice. Please try again.\n");
}
}
return 0;
}
