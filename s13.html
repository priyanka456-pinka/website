#include <stdio.h>

int main() {
char str[100], reversed[100];
int length = 0, i;

// Taking input for the string
printf("Enter a string: ");
fgets(str, sizeof(str), stdin);

// Calculate the length of the string manually
while (str[length] != '\0') {
// Stop counting if we hit a newline character
if (str[length] == '\n') {
break;
}
length++;
}

// Copy the reverse of the string into the 'reversed' array
for (i = 0; i < length; i++) {
reversed[i] = str[length - 1 - i];
}

// Add the null terminator at the end of the reversed string
reversed[length] = '\0';

// Output the reversed string
printf("Reversed string: %s\n", reversed);

return 0;
}
