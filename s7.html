CODE

#include <stdio.h>

int main() {
char str1[100], str2[100];
int i, isEqual = 1;

// Taking input for the first string
printf("Enter the first string: ");
fgets(str1, sizeof(str1), stdin);

// Taking input for the second string
printf("Enter the second string: ");
fgets(str2, sizeof(str2), stdin);

// Comparing the two strings character by character
for (i = 0; str1[i] != '\0' || str2[i] != '\0'; i++) {
// If characters are different, set isEqual to 0 and break the loop
if (str1[i] != str2[i]) {
isEqual = 0;
break;
}
}

// Adjust for newline character at the end (fgets includes it)
if (isEqual && str1[i-1] == '\n' && str2[i-1] == '\n') {
isEqual = 1;
} else {
isEqual = 0;
}

// Output the result
if (isEqual) {
printf("The strings are equal.\n");
} else {
printf("The strings are not equal.\n");
}

return 0;
}
