CODE

#include <stdio.h>
#include <string.h>

#define MAX_LENGTH 10 // Maximum length of the input string

int main() {
char str[MAX_LENGTH + 1]; // Array to hold the input string (including the null terminator)
int asciiValues[MAX_LENGTH]; // Array to store the ASCII values

// Input: Get the string from the user
printf("Enter a string (up to 10 characters): ");
fgets(str, sizeof(str), stdin);

// Remove the newline character if present
str[strcspn(str, "\n")] = '\0';
int length = strlen(str); // Get the actual length of the input string

// Convert each character to its ASCII value and store in the integer array
for (int i = 0; i < length; i++) {
asciiValues[i] = (int)str[i];
}

// Output: Display the ASCII values
printf("The ASCII values of the characters are:\n");
for (int i = 0; i < length; i++) {
printf("Character: %c, ASCII Value: %d\n", str[i], asciiValues[i]);
}
return 0;
}
