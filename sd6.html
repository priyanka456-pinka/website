CODE

#include <stdio.h>

int main() {
int n;

printf("Enter the number of elements in the array: ");
scanf("%d", &n);

int arr[n];

printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
printf("Element %d: ", i + 1);
scanf("%d", &arr[i]);
}

printf("Positive numbers in the array:\n");
for (int i = 0; i < n; i++) {
if (arr[i] > 0) {
printf("%d ", arr[i]);
}
}
printf("\n");

return 0;
}
