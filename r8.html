#include <stdio.h>

// Function to calculate the sum of elements in the array using recursion
int sumOfArray(int arr[], int n) {
// Base case: if there are no elements, return 0
if (n <= 0) {
return 0;
} else {
// Recursive case: add the last element to the sum of the remaining elements
return arr[n - 1] + sumOfArray(arr, n - 1);
}
}

// Function to calculate the arithmetic mean
float arithmeticMean(int arr[], int n) {
// Calculate sum using the sumOfArray function
int sum = sumOfArray(arr, n);

// Return mean (handle case where n is 0 to avoid division by zero)
return (n > 0) ? (float)sum / n : 0;
}

int main() {
int n;

printf("Enter the number of elements in the array: ");
scanf("%d", &n);

if (n <= 0) {
printf("Please enter a positive integer.");
return 1; // Exit if the input is invalid
}

int arr[n];
printf("Enter %d integers:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &arr[i]);
}

float mean = arithmeticMean(arr, n);
printf("The arithmetic mean of the array is: %.2f\n", mean);

return 0;
}
