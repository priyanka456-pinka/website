CODE

#include <stdio.h>
#include <stdlib.h>
#define MAX_SIZE 100
typedef struct TwoStacks {
int arr[MAX_SIZE];
int top1;
int top2;
} TwoStacks;
void initialize(TwoStacks* stacks) {
stacks->top1 = -1;
stacks->top2 = MAX_SIZE;
}
int isFull1(TwoStacks* stacks) {
return stacks->top1 + 1 == stacks->top2;
}
int isEmpty1(TwoStacks* stacks) {
return stacks->top1 == -1;
}
void push1(TwoStacks* stacks, int item) {
if (isFull1(stacks)) {
printf("Stack 1 is full! Cannot push %d\n", item);
return;
}
stacks->arr[++stacks->top1] = item;
}
int pop1(TwoStacks* stacks) {
if (isEmpty1(stacks)) {
printf("Stack 1 is empty! Cannot pop\n");
return -1;
}
return stacks->arr[stacks->top1--];
}
int isFull2(TwoStacks* stacks) {
return stacks->top2 - 1 == stacks->top1;
}
int isEmpty2(TwoStacks* stacks) {
return stacks->top2 == MAX_SIZE;
}
void push2(TwoStacks* stacks, int item) {
if (isFull2(stacks)) {
printf("Stack 2 is full! Cannot push %d\n", item);
return;
}
stacks->arr[--stacks->top2] = item;
}
int pop2(TwoStacks* stacks) {
if (isEmpty2(stacks)) {
printf("Stack 2 is empty! Cannot pop\n");
return -1;
}
return stacks->arr[stacks->top2++];
}
void displayStack1(TwoStacks* stacks) {
if (isEmpty1(stacks)) {
printf("Stack 1 is empty\n");
return;
}
printf("Stack 1 elements: ");
for (int i = 0; i <= stacks->top1; i++) {
printf("%d ", stacks->arr[i]);
}
printf("\n");
}
void displayStack2(TwoStacks* stacks) {
if (isEmpty2(stacks)) {
printf("Stack 2 is empty\n");
return;
}
printf("Stack 2 elements: ");
for (int i = MAX_SIZE - 1; i >= stacks->top2; i--) {
printf("%d ", stacks->arr[i]);
}
printf("\n");
}
int main() {
TwoStacks stacks;
initialize(&stacks);
push1(&stacks, 10);
push1(&stacks, 20);
push1(&stacks, 30);
displayStack1(&stacks);
push2(&stacks, 40);
push2(&stacks, 50);
push2(&stacks, 60);
displayStack2(&stacks);
printf("Popped from Stack 1: %d\n", pop1(&stacks));
printf("Popped from Stack 2: %d\n", pop2(&stacks));
displayStack1(&stacks);
displayStack2(&stacks);
return 0;
}
