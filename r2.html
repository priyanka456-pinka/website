#include <stdio.h>

// Function to calculate Fibonacci using recursion
int fibonacci(int n) {
if (n <= 0) {
return 0; // Fibonacci of 0 is 0
} else if (n == 1) {
return 1; // Fibonacci of 1 is 1
} else {
return fibonacci(n - 1) + fibonacci(n - 2); // Recursive case
}
}

int main() {
int num;
printf("Enter the position (n) to calculate the Fibonacci number: ");
scanf("%d", &num);

if (num < 0) {
printf("Fibonacci of a negative position is not defined.\n");
} else {
int result = fibonacci(num);
printf("Fibonacci number at position %d is: %d\n", num, result);
}

return 0;
}

OUTPUT

Enter the position (n) to calculate the Fibonacci number: 6
Fibonacci number at position 6 is: 8
