CODE

#include <stdio.h>

int main() {
int rows, cols;

// Ask for the size of the matrix
printf("Enter the number of rows: ");
scanf("%d", &rows);
printf("Enter the number of columns: ");
scanf("%d", &cols);

int matrix[rows][cols]; // Declare the original matrix
int copy[rows][cols]; // Declare the copy matrix
int i, j;
// Input the matrix elements
printf("Enter the elements of the matrix:\n");
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
printf("Element [%d][%d]: ", i, j);
scanf("%d", &matrix[i][j]);
}
}

// Copy the matrix to another matrix
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
copy[i][j] = matrix[i][j];
}
}

// Display the copied matrix
printf("\nThe copied matrix is:\n");
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
printf("%d ", copy[i][j]);
}
printf("\n");
}

return 0;
}
