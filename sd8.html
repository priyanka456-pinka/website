#include <stdio.h>
#include <limits.h>

int main() {
int n;

printf("Enter the number of elements in the array: ");
scanf("%d", &n);

int arr[n];

printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
printf("Element %d: ", i + 1);
scanf("%d", &arr[i]);
}

int max = INT_MIN;
int min = INT_MAX;

for (int i = 0; i < n; i++) {
if (arr[i] > max) {
max = arr[i];
}
if (arr[i] < min) {
min = arr[i];
}
}

printf("Maximum value in the array: %d\n", max);
printf("Minimum value in the array: %d\n", min);

return 0;
}
