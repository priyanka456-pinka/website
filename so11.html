#include <stdio.h>
#include <string.h>
#include <time.h>
#define MAX 100
#define MAX_LENGTH 100
void selectionSort(char arr[][MAX_LENGTH], int n) {
int i, j, minIndex;
char temp[MAX_LENGTH];
for (i = 0; i < n - 1; i++) {
minIndex = i;
for (j = i + 1; j < n; j++) {
if (strcmp(arr[j], arr[minIndex]) < 0) {
minIndex = j;
}
}
if (minIndex != i) {
strcpy(temp, arr[i]);
strcpy(arr[i], arr[minIndex]);
strcpy(arr[minIndex], temp);
}
}
}
int main() {
char arr[MAX][MAX_LENGTH];
int n, i;
clock_t start, end;
double time_taken;
printf("Enter the number of strings: ");
scanf("%d", &n);
getchar();
printf("Enter %d strings:\n", n);
for (i = 0; i < n; i++) {
fgets(arr[i], MAX_LENGTH, stdin);
arr[i][strcspn(arr[i], "\n")] = '\0';
}
start = clock();
selectionSort(arr, n);
end = clock();
time_taken = ((double)(end - start)) / CLOCKS_PER_SEC;
printf("Sorted array:\n");
for (i = 0; i < n; i++) {
printf("%s\n", arr[i]);
}
printf("Time taken for sorting: %f seconds\n", time_taken);
printf("Best case time complexity: O(n^2)\n");
printf("Worst case time complexity: O(n^2)\n");
return 0;
}
