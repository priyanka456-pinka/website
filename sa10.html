#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int compare(const void *a, const void *b) {
return (*(int *)a - *(int *)b);
}

int binarySearch(int arr[], int n, int target) {
int left = 0;
int right = n - 1;

while (left <= right) {
int mid = left + (right - left) / 2;

if (arr[mid] == target) {
return mid;
} else if (arr[mid] < target) {
left = mid + 1;
} else {
right = mid - 1;
}
}
return -1;
}

int main() {
int n;

printf("Enter the number of elements in the array: ");
scanf("%d", &n);

int arr[n];

srand(time(NULL));
for (int i = 0; i < n; i++) {
arr[i] = rand() % 100;
}

qsort(arr, n, sizeof(int), compare);

int bestCaseTarget = arr[n / 2];
clock_t startTime = clock();
int bestIndex = binarySearch(arr, n, bestCaseTarget);
clock_t endTime = clock();
double bestCaseTime = (double)(endTime - startTime) / CLOCKS_PER_SEC;

int worstCaseTarget = 150;
startTime = clock();
int worstIndex = binarySearch(arr, n, worstCaseTarget);
endTime = clock();
double worstCaseTime = (double)(endTime - startTime) / CLOCKS_PER_SEC;

printf("Best Case: Found %d at index %d in %f seconds\n", bestCaseTarget, bestIndex, bestCaseTime);
printf("Worst Case: Tried to find %d, result index %d in %f seconds\n", worstCaseTarget, worstIndex, worstCaseTime);

return 0;
}
