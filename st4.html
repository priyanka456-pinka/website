CODE

#include <stdio.h>
#include <stdlib.h>
typedef struct Node {
char data;
struct Node* next;
} Node;
typedef struct Stack {
Node* top;
int maxSize;
int currentSize;
} Stack;
void initialize(Stack* stack, int maxSize) {
stack->top = NULL;
stack->maxSize = maxSize;
stack->currentSize = 0;
}
int isEmpty(Stack* stack) {
return stack->top == NULL;
}
int isFull(Stack* stack) {
return stack->currentSize >= stack->maxSize;
}
void push(Stack* stack, char item) {
if (isFull(stack)) {
printf("Stack overflow! Cannot push '%c'\n", item);
return;
}
Node* newNode = (Node*)malloc(sizeof(Node));
if (newNode == NULL) {
printf("Memory allocation failed! Cannot push '%c'\n", item);
return;
}
newNode->data = item;
newNode->next = stack->top;
stack->top = newNode;
stack->currentSize++;
printf("Pushed '%c' onto the stack\n", item);
}
char pop(Stack* stack) {
if (isEmpty(stack)) {
printf("Stack underflow! Cannot pop from the stack\n");
return '\0';
}
Node* temp = stack->top;
char poppedValue = temp->data;
stack->top = stack->top->next;
free(temp);
stack->currentSize--;
return poppedValue;
}
char peek(Stack* stack) {
if (isEmpty(stack)) {
printf("Stack is empty! Cannot peek\n");
return '\0';
}
return stack->top->data;
}
int main() {
Stack stack;
int maxSize;
printf("Enter the maximum size of the stack: ");
scanf("%d", &maxSize);
initialize(&stack, maxSize);
char choice;
char item;
while (1) {
printf("\nStack Operations:\n");
printf("1. Push\n");
printf("2. Pop\n");
printf("3. Peek\n");
printf("4. Check if Empty\n");
printf("5. Check if Full\n");
printf("6. Exit\n");
printf("Enter your choice: ");
scanf(" %c", &choice);
switch (choice) {
case '1':
printf("Enter a character to push: ");
scanf(" %c", &item);
push(&stack, item);
break;
case '2':
item = pop(&stack);
if (item != '\0') {
printf("Popped '%c' from the stack\n", item);
}
break;
case '3':
item = peek(&stack);
if (item != '\0') {
printf("Top element is: '%c'\n", item);
}
break;
case '4':
if (isEmpty(&stack)) {
printf("Stack is empty.\n");
} else {
printf("Stack is not empty.\n");
}
break;
case '5':
if (isFull(&stack)) {
printf("Stack is full.\n");
} else {
printf("Stack is not full.\n");
}
break;
case '6':
printf("Exiting the program. Goodbye!\n");
return 0;
default:
printf("Invalid choice. Please try again.\n");
}
}
return 0;
}
