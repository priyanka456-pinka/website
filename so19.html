#include <stdio.h>
#include <string.h>
#include <time.h>
#define MAX_STRING_LENGTH 100
#define MAX_ARRAY_SIZE 100
void swap(char *str1, char *str2) {
char temp[MAX_STRING_LENGTH];
strcpy(temp, str1);
strcpy(str1, str2);
strcpy(str2, temp);
}
int partition(char arr[][MAX_STRING_LENGTH], int low, int high) {
char *pivot = arr[high];
int i = (low - 1);
for (int j = low; j < high; j++) {
if (strcmp(arr[j], pivot) <= 0) {
i++;
swap(arr[i], arr[j]);
}
}
swap(arr[i + 1], arr[high]);
return (i + 1);
}
void quickSortIterative(char arr[][MAX_STRING_LENGTH], int n) {
int stack[n];
int top = -1;
stack[++top] = 0;
stack[++top] = n - 1;
while (top >= 0) {
int high = stack[top--];
int low = stack[top--];
int pivotIndex = partition(arr, low, high);
if (pivotIndex - 1 > low) {
stack[++top] = low;
stack[++top] = pivotIndex - 1;
}
if (pivotIndex + 1 < high) {
stack[++top] = pivotIndex + 1;
stack[++top] = high;
}
}
}
void printArray(char arr[][MAX_STRING_LENGTH], int n) {
for (int i = 0; i < n; i++) {
printf("%s ", arr[i]);
}
printf("\n");
}
int main() {
char arr[MAX_ARRAY_SIZE][MAX_STRING_LENGTH];
int n;
printf("Enter the number of strings: ");
scanf("%d", &n);
getchar();
printf("Enter the strings:\n");
for (int i = 0; i < n; i++) {
fgets(arr[i], sizeof(arr[i]), stdin);
arr[i][strcspn(arr[i], "\n")] = '\0';
}
clock_t start, end;
double cpu_time_used;
start = clock();
quickSortIterative(arr, n);
end = clock();
cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
printf("Sorted array of strings:\n");
printArray(arr, n);
printf("Time taken for sorting: %f seconds\n", cpu_time_used);
printf("\nComplexity Analysis:\n");
printf("Worst Case Complexity: O(n^2)\n");
printf("Best Case Complexity: O(n log n)\n");
return 0;
}
