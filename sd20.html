#include <stdio.h>
int trap(int height[], int size) {
if (size <= 2) return 0;
int left = 0, right = size - 1;
int left_max = 0, right_max = 0;
int water_trapped = 0;
while (left < right) {
if (height[left] < height[right]) {
if (height[left] >= left_max) {
left_max = height[left];
} else {
water_trapped += left_max - height[left];
}
left++;
} else {
if (height[right] >= right_max) {
right_max = height[right];
} else {
water_trapped += right_max - height[right];
}
right--;
}
}
return water_trapped;
}
int main() {
int n;
printf("Enter the number of elements in the elevation map: ");
scanf("%d", &n);
int height[n];
printf("Enter %d non-negative integers representing the elevation map:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &height[i]);
}
int result = trap(height, n);
printf("Total water trapped: %d\n", result);
return 0;
}
