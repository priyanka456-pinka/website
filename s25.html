#include <stdio.h>

// Function to remove leading and trailing spaces
void trimSpaces(char *str) {
int start = 0; // Index for the first non-space character
int end = 0; // Index for the last non-space character
int i = 0;
// Find the first non-space character from the beginning
while (str[start] == ' ') {
start++;
}
// Find the last non-space character from the end
end = start;
while (str[end] != '\0') {
end++;
}
end--; // Move to the last valid character
// Move backward to find the last non-space character
while (end >= 0 && str[end] == ' ') {
end--;
}
// If there are no non-space characters, the string is empty
if (start > end) {
str[0] = '\0';
return;
}
// Shift the characters to the beginning of the string
i = 0;
while (start <= end) {
str[i++] = str[start++];
}
// Null-terminate the trimmed string
str[i] = '\0';
}

int main() {
char str[1000]; // Array to hold the input string
// Input: Get the string from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin); // Read the input string
// Remove the newline character if it exists
int length = 0;
while (str[length] != '\0') {
if (str[length] == '\n') {
str[length] = '\0';
break;
}
length++;
}
// Trim the spaces from both ends
trimSpaces(str);
// Output the trimmed string
printf("The trimmed string is: \"%s\"\n", str);
return 0;
}
