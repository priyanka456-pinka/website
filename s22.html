#include <stdio.h>
#include <string.h>
void sortString(char str[]) {
int n = strlen(str);
char temp;
// Sort the string using bubble sort
for (int i = 0; i < n - 1; i++) {
for (int j = i + 1; j < n; j++) {
if (str[i] > str[j]) {
// Swap the characters
temp = str[i];
str[i] = str[j];
str[j] = temp;
}
}
}
}
int main() {
char str[100];
// Get input from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin);
// Remove the newline character if present
str[strcspn(str, "\n")] = 0;
// Sort the string
sortString(str);
// Print the sorted string
printf("Sorted string: %s\n", str);
return 0;
}
