#include <stdio.h>
#include <string.h>

#define SCREEN_WIDTH 80 // Assume a screen width of 80 characters

// Function to display the string with specified alignment
void displayAlignedText(char *str, int alignment) {
int length = strlen(str); // Get the length of the string
// Remove the newline character at the end, if any
if (str[length - 1] == '\n') {
str[length - 1] = '\0';
length--;
}
// Display the string based on the chosen alignment
switch (alignment) {
case 1: // Left alignment
printf("%s\n", str);
break;
case 2: // Center alignment
{
int padding = (SCREEN_WIDTH - length) / 2;
if (padding > 0) {
printf("%*s%s\n", padding, "", str); // Add padding spaces before the string
} else {
printf("%s\n", str); // If string is longer than screen width, print as is
}
}
break;
case 3: // Right alignment
{
int padding = SCREEN_WIDTH - length;
if (padding > 0) {
printf("%*s%s\n", padding, "", str); // Add padding spaces before the string
} else {
printf("%s\n", str); // If string is longer than screen width, print as is
}
}
break;
default:
printf("Invalid alignment option. Please choose 1, 2, or 3.\n");
break;
}
}

int main() {
char str[1000]; // Array to hold the input string
int alignment; // Variable to hold the alignment option
// Input: Get the string from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin); // Read the input string
// Input: Get the alignment option from the user
printf("Choose the alignment option (1 for Left, 2 for Center, 3 for Right): ");
scanf("%d", &alignment);
// Display the aligned text
displayAlignedText(str, alignment);
return 0;
}
