#include <stdio.h>

int main() {
int matrix1[2][2], matrix2[2][2], product[2][2];
int i, j, k;

// Input the first matrix
printf("Enter elements of the first 2x2 matrix:\n");
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
printf("Element [%d][%d]: ", i, j);
scanf("%d", &matrix1[i][j]);
}
}

// Input the second matrix
printf("Enter elements of the second 2x2 matrix:\n");
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
printf("Element [%d][%d]: ", i, j);
scanf("%d", &matrix2[i][j]);
}
}

// Initialize the product matrix to 0
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
product[i][j] = 0;
}
}

// Multiply the matrices
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
for (k = 0; k < 2; k++) {
product[i][j] += matrix1[i][k] * matrix2[k][j];
}
}
}

// Display the result
printf("\nThe product of the two matrices is:\n");
for (i = 0; i < 2; i++) {
for (j = 0; j < 2; j++) {
printf("%d ", product[i][j]);
}
printf("\n");
}

return 0;
}
