#include <stdio.h>

int main() {
char str[100];
int i = 0;

// Taking input for the string
printf("Enter a string: ");
fgets(str, sizeof(str), stdin);

// Print each character in a new line
while (str[i] != '\0') {
// Stop printing if we hit a newline character
if (str[i] == '\n') {
break;
}
// Print the current character
printf("%c\n", str[i]);
i++;
}

return 0;
}
