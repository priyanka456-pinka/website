#include <stdio.h>

// Function to find the start index of the second word
int findSecondWordStartIndex(char *str) {
int i = 0;
// Skip leading spaces
while (str[i] == ' ') {
i++;
}
// Skip the first word
while (str[i] != ' ' && str[i] != '\0') {
i++;
}
// Skip spaces between the first and second word
while (str[i] == ' ') {
i++;
}
// Check if there is a second word
if (str[i] == '\0') {
return -1; // No second word found
}
// Return the start index of the second word
return i;
}

int main() {
char str[1000]; // Array to hold the input string
// Input: Get the string from the user
printf("Enter a string: ");
fgets(str, sizeof(str), stdin); // Read the input string
// Find the start index of the second word
int index = findSecondWordStartIndex(str);
// Output the results
if (index == -1) {
printf("No second word found in the input string.\n");
} else {
printf("The start index of the second word is: %d\n", index);
}
return 0;
}
