#include <stdio.h>
void printCombinations(int arr[], int n, int index, int data[], int i) {
if (index > 0) {
for (int j = 0; j < index; j++) {
printf("%d ", data[j]);
}
printf("\n");
}
for (int j = i; j < n; j++) {
data[index] = arr[j];
printCombinations(arr, n, index + 1, data, j + 1);
}
}
int main() {
int n;
printf("Enter the number of elements in the array: ");
scanf("%d", &n);
int arr[n];
printf("Enter %d elements:\n", n);
for (int i = 0; i < n; i++) {
scanf("%d", &arr[i]);
}
int data[n];
printf("All combinations of the array elements are:\n");
printCombinations(arr, n, 0, data, 0);
return 0;
}
