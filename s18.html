CODE

#include <stdio.h>
int main() {
char str[100]; // Array to hold the input string
int number = 0; // Variable to hold the converted integer
int i = 0; // Loop counter
int sign = 1; // Variable to store the sign of the number (1 for positive, -1 for negative)
// Input: string from the user
printf("Enter a string to convert to an integer: ");
fgets(str, sizeof(str), stdin); // Use fgets to read a line of text
// Check for a negative sign
if (str[0] == '-') {
sign = -1; // Set sign to -1 for negative numbers
i++; // Start from the next character
} else if (str[0] == '+') {
i++; // Ignore the plus sign and start from the next character
}
// Convert each character to an integer
for (; str[i] != '\0'; i++) {
// Check if the character is a digit
if (str[i] >= '0' && str[i] <= '9') {
number = number * 10 + (str[i] - '0'); // Build the integer value
} else if (str[i] == '\n') {
// Stop at newline character (if using fgets)
break;
} else {
// Handle non-numeric characters
printf("Error: Invalid character '%c' in input.\n", str[i]);
return 1; // Exit with error code
}
}
// Apply the sign to the number
number *= sign;
// Output the result
printf("The integer value is: %d\n", number);
return 0;
}

OUTPUT

Enter a string to convert to an integer: 125
The integer value is: 125
