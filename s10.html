CODE

#include <stdio.h>

int main() {
char str[100];
int index, length = 0;

// Taking input for the string
printf("Enter a string: ");
fgets(str, sizeof(str), stdin);

// Calculate the length of the string manually
while (str[length] != '\0') {
// Stop counting if we hit a newline character
if (str[length] == '\n') {
break;
}
length++;
}

// Taking input for the index number
printf("Enter the index number: ");
scanf("%d", &index);

// Check if the index is within the valid range
if (index >= 0 && index < length) {
// Printing the character at the specified index
printf("Character at index %d: %c\n", index, str[index]);
} else {
// If the index is out of range
printf("Invalid index! Please enter a number between 0 and %d.\n", length - 1);
}

return 0;
}
