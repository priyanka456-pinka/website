scanf("%d", &matrix[i][j]);
}
}
// Display the matrix
printf("\nThe 4x2 matrix is:\n");
for (i = 0; i < 4; i++) {
for (j = 0; j < 2; j++) {
printf("%d ", matrix[i][j]);
}
printf("\n"); // New line after each row
}
return 0;
}
