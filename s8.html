#include <stdio.h>

int main() {
char str1[100], str2[100], result[200];
int i = 0, j = 0;

// Taking input for the first string
printf("Enter the first string: ");
fgets(str1, sizeof(str1), stdin);

// Taking input for the second string
printf("Enter the second string: ");
fgets(str2, sizeof(str2), stdin);

// Copying the first string into the result
while (str1[i] != '\0') {
if (str1[i] == '\n') {
i++; // Skip the newline character
continue;
}
result[j] = str1[i];
i++;
j++;
}

// Reset i for the second string
i = 0;

// Copying the second string into the result
while (str2[i] != '\0') {
if (str2[i] == '\n') {
i++; // Skip the newline character
continue;
}
result[j] = str2[i];
i++;
j++;
}

// Add the null terminator at the end of the result string
result[j] = '\0';

// Output the concatenated string
printf("Concatenated string: %s\n", result);

return 0;
}
