#include <stdio.h>
#include <stdlib.h>
#define MAX_VERTICES 10
// Function to perform DFS traversal
void DFS(int graph[MAX_VERTICES][MAX_VERTICES], int visited[], int vertex, int numVertices) {

visited[vertex] = 1;
printf("%d ", vertex);

for (int i = 0; i < numVertices; i++) {

if (graph[vertex][i] == 1 && !visited[i]) {
DFS(graph, visited, i, numVertices);
}
}
}
int main() {
int numVertices;
int graph[MAX_VERTICES][MAX_VERTICES] = {0};

printf("Enter the number of vertices (max %d): ", MAX_VERTICES);
scanf("%d", &numVertices);


printf("Enter the adjacency matrix (%dx%d):\n", numVertices, numVertices);
for (int i = 0; i < numVertices; i++) {
for (int j = 0; j < numVertices; j++) {
scanf("%d", &graph[i][j]);
}
}

int visited[MAX_VERTICES] = {0};

printf("Depth First Search (DFS) traversal starting from vertex 0:\n");
DFS(graph, visited, 0, numVertices);

printf("\n");

return 0;
}
