#include <stdio.h>

// Function to calculate the sum of natural numbers using recursion
int sumOfNaturalNumbers(int n) {
if (n == 0) {
return 0; // Base case: sum of natural numbers up to 0 is 0
} else {
return n + sumOfNaturalNumbers(n - 1); // Recursive case
}
}

int main() {
int num = 10; // Calculate the sum of natural numbers up to 10
int result = sumOfNaturalNumbers(num);
printf("The sum of natural numbers till %d is: %d\n", num, result);
return 0;
}

OUTPUT

The sum of natural numbers till 10 is: 55
