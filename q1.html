CODE

#include <stdio.h>
#include <stdlib.h>

#define MAX 5 // Define the maximum size of the queue

typedef struct {
int items[MAX];
int front;
int rear;
int size;
} Queue;

// Function to initialize the queue
void initQueue(Queue *q) {
q->front = 0;
q->rear = -1;
q->size = 0;
}

// Function to check if the queue is empty
int isEmpty(Queue *q) {
return q->size == 0;
}

// Function to check if the queue is full
int isFull(Queue *q) {
return q->size == MAX;
}

// Function to add an item to the queue
void enqueue(Queue *q, int value) {
if (isFull(q)) {
printf("Queue is full. Cannot enqueue %d.\n", value);
return;
}
q->rear = (q->rear + 1) % MAX; // Circular increment
q->items[q->rear] = value;
q->size++;
printf("Enqueued: %d\n", value);
}

// Function to remove an item from the queue
int dequeue(Queue *q) {
if (isEmpty(q)) {
printf("Queue is empty. Cannot dequeue.\n");
return -1; // Return -1 to indicate the queue is empty
}
int value = q->items[q->front];
q->front = (q->front + 1) % MAX; // Circular increment
q->size--;
printf("Dequeued: %d\n", value);
return value;
}

// Function to get the front item of the queue
int getFront(Queue *q) {
if (isEmpty(q)) {
printf("Queue is empty. No front item.\n");
return -1; // Return -1 to indicate the queue is empty
}
return q->items[q->front];
}

// Function to display the elements of the queue
void display(Queue *q) {
if (isEmpty(q)) {
printf("Queue is empty.\n");
return;
}
printf("Queue elements: ");
for (int i = 0; i < q->size; i++) {
int index = (q->front + i) % MAX;
printf("%d ", q->items[index]);
}
printf("\n");
}

// Main function to demonstrate queue operations
int main() {
Queue q;
initQueue(&q);

enqueue(&q, 10);
enqueue(&q, 20);
enqueue(&q, 30);
display(&q); // Output: Queue elements: 10 20 30

dequeue(&q);
display(&q); // Output: Queue elements: 20 30

printf("Front element: %d\n", getFront(&q)); // Output: Front element: 20

enqueue(&q, 40);
enqueue(&q, 50);
enqueue(&q, 60); // This should print "Queue is full. Cannot enqueue."
display(&q); // Output: Queue elements: 20 30 40 50

return 0;
}
