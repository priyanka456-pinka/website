CODE

#include <stdio.h>
int main() {

int rows, cols;
float scalar;
printf("Enter the number of rows: ");
scanf("%d", &rows);
printf("Enter the number of columns: ");
scanf("%d", &cols);
float matrix[rows][cols]; // Declare the original matrix
float result[rows][cols]; // Declare the result matrix
int i, j;
// Input the matrix elements
printf("Enter the elements of the matrix:\n");
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
printf("Element [%d][%d]: ", i, j);
scanf("%f", &matrix[i][j]);
}
}
// Input the scalar value
printf("Enter the scalar value: ");
scanf("%f", &scalar);
// Multiply the matrix by the scalar
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
result[i][j] = matrix[i][j] * scalar;
}
}
// Display the resulting matrix
printf("\nThe resulting matrix after multiplication by %.2f is:\n", scalar);
for (i = 0; i < rows; i++) {
for (j = 0; j < cols; j++) {
printf("%.2f ", result[i][j]);
}
printf("\n"); // New line after each row
}
return 0;
}
